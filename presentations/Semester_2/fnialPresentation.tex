%%%% Slide 3 -- Linear Algebra Crash Course (Refresher)
\begin{frame}
\frametitle{Linear Algebra Refresher}
\begin{columns}
\begin{column}{0.5\textwidth}
    \textbf{Dirac Notation}
    \begin{itemize}
        \item Ket: $\ket{\psi} = \begin{pmatrix} \alpha \\ \beta \end{pmatrix}$
        \item Bra: $\bra{\psi} = \begin{pmatrix} \alpha^* & \beta^* \end{pmatrix}$
        \item Inner Product: $\braket{\phi}{\psi}$
    \end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
    \textbf{Tensor Products}
    Used to represent multi-qubit systems:
    \[ \ket{\psi} \otimes \ket{\phi} = \ket{\psi\phi} \]
    \textbf{Conjugate Transpose}
    \[ A^\dagger = (A^T)^* \]
    A gate $U$ is valid if $U^\dagger U = I$.
\end{column}
\end{columns}
\end{frame}

%%%% Slide 4 -- What is a Quantum Gate?
\begin{frame}
\frametitle{Quantum Gates: The Ideal Physics World}
\begin{block}{High-Level Definition}
    A quantum gate is a unitary transformation applied to a qubit state. In a "perfect" physics world, these operations are:
\end{block}
\begin{itemize}
    \item \textbf{Deterministic:} Always performs the exact rotation intended.
    \item \textbf{Isolated:} No interaction with the outside environment.
    \item \textbf{Reversible:} No information is lost during computation.
\end{itemize}

%%%% Slide 8 -- The Interface: CLI
\begin{frame}[fragile]
\frametitle{Command Line Interface (CLI)}
In order to bridge the gap between the high level Qiskit simulations and low-level C phsysics simulation I developed a CLI to simulate perfect quantum gates.
\begin{exampleblock}{Sample Usage}
\begin{verbatim}
$ qit> create q1 10 
$ qit> cnot q1 0 1 ## 0 is ctrl and 1 is targ
$ qit> entangle bp1 q1 2 3 ## bell pair
$ qit> measure q1 0 1 2 3
\end{verbatim}
\end{exampleblock}
This also allows for easy integration with the C-based simulator, where we can directly tell the CLI to read from the QIT output.
\end{frame}

%%%% Slide 9 -- Physics Simulation in C
\begin{frame}
\frametitle{Low-Level Simulation in C}

Schrodinger Equation:
\[ i \frac{d}{dt} |\psi(t)\rangle = \hat{H}(t) |\psi(t)\rangle \]
Single Qubit Hamiltonian:
\[ \hat{H}_{k} = \frac{1}{2} \delta \hat{\sigma}_z + \frac{1}{2} \Omega ( \cos(\phi) \hat{\sigma}_x + \sin(\phi) \hat{\sigma}_y ) \]
Cross-Resonance Hamiltonian:
\[ \hat{H}_{CR} = \frac{\Omega_{cr}}{2} ( \hat{\sigma}_z^{(c)} \otimes \hat{\sigma}_x^{(t)} ) \]

\frametitle{Takeaways and Conclusions}

    \item \textbf{Integration is Key:} Quantum computers won't replace clusters; they will act as specialized ``accelerators.''
    
    \item \textbf{Hardware vs. Simulation:} Simulation is essential for debugging QEC algorithms before running on expensive hardware.

        \item \textbf{Example:} Simulating a noisy CNOT gate to understand error rates.
    \end{itemize}

\end{itemize}
